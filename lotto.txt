	<head>
		<meta charset="utf-8">
		<meta name="viewport" content="width=device-width,initial-scale=1">
		<title> 수학객체(로또).html </title>
		<script>
			var list=[]; // 로또 번호 저장
			var i = 0; // 로또배열의 인덱스 사용
			var j;  // for문에 인덱스 사용
			var flag;  // 중복유무 -> 1이면 중복아님. 0이면 중복
			var k;
			
			for( k = 1 ; k <= 5 ; k++){
				i = 0; //   처음 부터 다시 추출하기 위해 인덱스 초기화
				list=[]; // 처음 부터 다시 추출하기 위해 배열 초기화
				while ( i <= 5 ){ // 로또번호를 다채우면 끝! i가 6이되면 종료
					list[i] = Math.floor(Math.random()*45)+1;
					flag = 1; // 새롭게 숫자를 받으면 flag를 초기화!
					for( j = 0 ; j <= list.length-2 ; j++){
						// 배열이 채워진 만큼 비교하기 위해서
						if( list[i] == list[j] ){ flag = 0;}
					}
					if( flag != 0 ) { i++;} // 중복아니면 i인덱스 증가!
				}
				
				var temp ;  // 작은수에서 큰수로 정렬
				for( i = 0 ; i <= list.length-2 ; i++){
					for( j = i+1 ; j <= list.length-1; j++){
						if( list[i] > list[j]){
							temp = list[i];
							list[i] = list[j];
							list[j] = temp;
						}
					}
				}
				document.write(list+"<br>");
			}
		</script>
